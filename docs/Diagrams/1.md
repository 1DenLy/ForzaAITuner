C4Container
    title Container Diagram for ForzaAITuner

    Person(driver, "Driver", "Player playing Forza Horizon")
    
    System_Boundary(tuner_system, "Forza AI Tuner System") {
        Container(game, "Forza Horizon Game", "External System", "Sends telemetry via UDP (Data Out)")
        
        Container(ingestion, "Ingestion Service (forza_core)", "Python, AsyncIO", "Listens to UDP packets, parses binary data, buffers, and writes to DB.")
        
        Container(db, "Time-Series Database (forza_db)", "PostgreSQL + TimescaleDB", "Stores raw telemetry (Hypertable) and relational tuning data.")
        
        Container(processor, "Analytics Worker (forza_processor)", "Python", "Reads raw data, aggregates stats, runs ML inference.")
        
        Container(admin, "Admin UI (forza_admin)", "pgAdmin4", "Database management and inspection.")
    }

    Rel(driver, game, "Drives")
    Rel(game, ingestion, "Sends Telemetry Packets", "UDP:5300", "60Hz, 324 bytes/packet")
    
    Rel(ingestion, db, "Batch Insert (COPY)", "TCP:5432", "Asyncpg")
    
    Rel(processor, db, "Read (Analysis) / Write (Insights)", "TCP:5432", "SQLAlchemy")
    
    Rel(admin, db, "Manage Schema & Data", "TCP:5432")
